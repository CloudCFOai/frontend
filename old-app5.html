// OAuth 2.0 Security Implementation class OAuthSecurityManager { private readonly encryptionKey = process.env.OAUTH_ENCRYPTION_KEY; private readonly jwtSecret = process.env.JWT_SECRET; // PKCE (Proof Key for Code Exchange) implementation async generatePKCEChallenge(): Promise { const verifier = crypto.randomBytes(32).toString('base64url'); const challenge = crypto .createHash('sha256') .update(verifier) .digest('base64url'); return { verifier, challenge }; } // Secure token storage with encryption async storeTokens(userId: string, tokens: OAuthTokens) { const encrypted = await this.encryptTokens(tokens); await db.oauthTokens.create({ userId, encryptedAccessToken: encrypted.accessToken, encryptedRefreshToken: encrypted.refreshToken, expiresAt: new Date(Date.now() + tokens.expiresIn * 1000), scope: tokens.scope, tokenType: 'Bearer' }); } // Token rotation for enhanced security async rotateTokens(userId: string, refreshToken: string) { const decrypted = await this.decryptToken(refreshToken); // Validate refresh token if (!this.isValidRefreshToken(decrypted)) { throw new SecurityError('Invalid refresh token'); } // Request new tokens const newTokens = await this.requestNewTokens(decrypted); // Invalidate old tokens await this.invalidateOldTokens(userId); // Store new tokens await this.storeTokens(userId, newTokens); return newTokens; } // Encryption using AES-256-GCM private async encryptTokens(tokens: OAuthTokens) { const algorithm = 'aes-256-gcm'; const iv = crypto.randomBytes(16); const cipher = crypto.createCipheriv( algorithm, Buffer.from(this.encryptionKey, 'hex'), iv ); const encrypted = { accessToken: this.encryptString(tokens.accessToken, cipher), refreshToken: this.encryptString(tokens.refreshToken, cipher) }; return encrypted; } }

// Gmail Integration Service class GmailIntegration { private readonly scopes = [ 'https://www.googleapis.com/auth/gmail.readonly', 'https://www.googleapis.com/auth/gmail.modify' ]; async searchReceipts(auth: OAuth2Client): Promise { const gmail = google.gmail({ version: 'v1', auth }); // Advanced search query for receipts const query = [ 'has:attachment', '(subject:receipt OR subject:invoice OR subject:order)', 'from:(amazon OR uber OR starbucks OR "no-reply")', 'newer_than:1y', 'filename:(pdf OR jpg OR png)' ].join(' '); const messages = await gmail.users.messages.list({ userId: 'me', q: query, maxResults: 500 }); const receipts = []; for (const message of messages.data.messages || []) { const receipt = await this.processMessage(gmail, message.id); if (receipt) receipts.push(receipt); } return receipts; } async processMessage(gmail: any, messageId: string): Promise { const message = await gmail.users.messages.get({ userId: 'me', id: messageId, format: 'full' }); const attachments = await this.extractAttachments(gmail, message); const metadata = this.extractMetadata(message); if (attachments.length > 0) { return { source: 'gmail', messageId, subject: metadata.subject, sender: metadata.from, date: metadata.date, attachments, confidence: this.calculateConfidence(metadata) }; } return null; } private async extractAttachments(gmail: any, message: any): Promise { const attachments = []; const parts = message.data.payload.parts || []; for (const part of parts) { if (part.filename && part.body.attachmentId) { const attachment = await gmail.users.messages.attachments.get({ userId: 'me', messageId: message.data.id, id: part.body.attachmentId }); attachments.push({ filename: part.filename, mimeType: part.mimeType, size: part.body.size, data: attachment.data.data // Base64 encoded }); } } return attachments; } }

Multi-API Carbon Integration

// Carbon Calculation Service class CarbonCalculationService { private apis = { epa: new EPAEmissionsAPI(), ecoinvent: new EcoinventAPI(), transport: new TransportEmissionsAPI(), materials: new MaterialsDatabase() }; async calculateProductCarbon( product: Product, invoices: Invoice[] ): Promise { // Categorize invoices by scope const scopedInvoices = this.categorizeByScope(invoices); // Calculate Scope 1: Direct emissions const scope1 = await this.calculateScope1(scopedInvoices.direct); // Calculate Scope 2: Energy indirect const scope2 = await this.calculateScope2(scopedInvoices.energy); // Calculate Scope 3: Other indirect const scope3 = await this.calculateScope3(scopedInvoices.indirect); return { product, total: scope1.total + scope2.total + scope3.total, perUnit: (scope1.total + scope2.total + scope3.total) / product.units, breakdown: { scope1, scope2, scope3 }, methodology: 'GHG Protocol Product Standard', confidence: this.calculateConfidence([scope1, scope2, scope3]), timestamp: new Date() }; } private async calculateScope2(energyInvoices: Invoice[]): Promise { const emissions = []; for (const invoice of energyInvoices) { // Get grid emission factor from EPA const gridFactor = await this.apis.epa.getGridEmissionFactor({ location: invoice.vendor.location, year: invoice.date.getFullYear() }); // Calculate emissions const kwh = this.extractEnergyUsage(invoice); const co2e = kwh * gridFactor.factor; emissions.push({ source: invoice.vendor.name, amount: co2e, unit: 'kg CO2e', factor: gridFactor }); } return { total: emissions.reduce((sum, e) => sum + e.amount, 0), emissions, methodology: 'EPA eGRID factors' }; } }
