Database Schema

Core Tables (Shared with n8n)

-- Users table (managed by Supabase Auth) CREATE TABLE users ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), email TEXT UNIQUE NOT NULL, created_at TIMESTAMPTZ DEFAULT NOW(), subscription_tier TEXT DEFAULT 'free', settings JSONB DEFAULT '{}'::jsonb ); -- Documents table CREATE TABLE documents ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), user_id UUID REFERENCES users(id), filename TEXT NOT NULL, file_url TEXT NOT NULL, source TEXT NOT NULL, status TEXT DEFAULT 'uploading', n8n_webhook_id TEXT, created_at TIMESTAMPTZ DEFAULT NOW(), processed_at TIMESTAMPTZ, metadata JSONB DEFAULT '{}'::jsonb ); -- Transactions table (shared with n8n) CREATE TABLE transactions ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), document_id UUID REFERENCES documents(id), user_id UUID REFERENCES users(id), vendor_id UUID REFERENCES vendors(id), amount DECIMAL(10,2) NOT NULL, currency TEXT DEFAULT 'USD', transaction_date DATE NOT NULL, category_id UUID REFERENCES categories(id), extracted_data JSONB, created_at TIMESTAMPTZ DEFAULT NOW() ); -- Knowledge Base patterns CREATE TABLE kb_patterns ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), vendor_id UUID REFERENCES vendors(id), pattern_data JSONB NOT NULL, confidence_score DECIMAL(3,2), usage_count INTEGER DEFAULT 0, last_used TIMESTAMPTZ, created_at TIMESTAMPTZ DEFAULT NOW() ); -- Real-time processing events CREATE TABLE processing_events ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), document_id UUID REFERENCES documents(id), event_type TEXT NOT NULL, event_data JSONB, created_at TIMESTAMPTZ DEFAULT NOW() );

Real-time Processing Architecture

Socket.io Server Implementation

// server/socket.ts import { Server } from 'socket.io'; import Redis from 'ioredis'; const redis = new Redis(); const pubClient = new Redis(); const subClient = new Redis(); export const initializeSocketServer = (httpServer: any) => { const io = new Server(httpServer, { cors: { origin: process.env.FRONTEND_URL, credentials: true } }); // Subscribe to n8n processing events subClient.subscribe('document:processing'); subClient.on('message', (channel, message) => { const event = JSON.parse(message); // Emit to specific document room io.to(`doc:${event.documentId}`).emit('processing', { status: event.status, data: event.data, timestamp: event.timestamp }); }); io.on('connection', (socket) => { // Join document-specific room socket.on('subscribe:document', (documentId) => { socket.join(`doc:${documentId}`); }); // Handle user authentication socket.on('auth', async (token) => { const user = await verifyToken(token); socket.data.userId = user.id; }); }); };

Security Implementation

Authentication Flow with Clerk

// middleware.ts import { authMiddleware } from '@clerk/nextjs'; export default authMiddleware({ publicRoutes: ['/', '/api/webhooks/n8n'], ignoredRoutes: ['/api/public'] }); // API Route Protection import { getAuth } from '@clerk/nextjs/server'; export async function POST(req: Request) { const { userId } = getAuth(req); if (!userId) { return new Response('Unauthorized', { status: 401 }); } // Rate limiting per user const rateLimitKey = `upload:${userId}`; const uploads = await redis.incr(rateLimitKey); if (uploads === 1) { await redis.expire(rateLimitKey, 3600); // 1 hour window } if (uploads > 100) { return new Response('Rate limit exceeded', { status: 429 }); } // Process upload... }

Performance Optimization

Key Optimizations

  • Image Optimization: Next.js Image component with automatic WebP conversion
  • Code Splitting: Dynamic imports for heavy components (charts, upload)
  • Caching Strategy:
    • Static assets: 1 year cache
    • API responses: 5 minute cache for reports
    • Knowledge base: 24 hour cache
  • Database Queries: Indexed on user_id, created_at, vendor_id
  • WebSocket Optimization: Room-based broadcasting, connection pooling
